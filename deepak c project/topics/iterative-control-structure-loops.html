2.Enter a number from the user and check that the number is divisible by any prime number or not ?"

#include <stdio.h>
void main() {
int num, prime, got = 0;
printf("Enter a number: ");
scanf("%d", &num);
for (int i = 2; i <= num / 2; i++) {
prime = 1;
for (int j = 2; j <= i / 2; j++) {
if (i % j == 0) { prime = 0; break; }
}
if (prime && num % i == 0) { got = 1; break; }
}
if (got)
printf("The number %d is divisible by a prime number\n", num);
else
printf("The number %d is not divisible by a prime number\n", num);
}

22.Enter a number from user and check it is Armstrong number or not ?

#include <stdio.h>
void main() {
int num, temp, rem, n = 0;
double result = 0.0;
printf("Enter a number: ");
scanf("%d", &num);
temp = num;
while (temp != 0) { temp /= 10; ++n; }
temp = num;
while (temp != 0) {
rem = temp % 10;
result += pow(rem, n);
temp /= 10;
}
if ((int)result == num) printf("%d is an Armstrong number.\n", num);
else printf("%d is not an Armstrong number.\n", num);
}
