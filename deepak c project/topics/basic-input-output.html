10.Write a program to store single quote and print that variable to show single quote on output screen ? 

#include  <stdio.h>
 #include <conio.h>
 void main() {
    char password[20];
    int i = 0;
    char ch;

    printf("Enter password: ");

    while(1) {
        ch = getch();  // Read a single character without echo
        if(ch == 13) { // Enter key ASCII = 13
            break;
        }
        password[i] = ch;
        i++;
        printf("*"); // Print * instead of actual character
    }
    password[i] = '\0'; // Null-terminate the string

    printf("\nYour password is: %s\n", password);
}
