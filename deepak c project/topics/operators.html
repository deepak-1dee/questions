11.Enter marks of students and calculate average of studentâ€™s marks ?

#include <stdio.h>
void main() {
    int n, i, marks, sum=0;
    printf("Enter number of students: ");
    scanf("%d", &n);
    for(i=1;i<=n;i++){
        printf("Enter marks of student %d: ", i);
        scanf("%d",&marks);
        sum += marks;
    }
    printf("Average marks: %.2f", (float)sum/n);
}

12.Calculate simple interest = (P x R x T)/ 100

#include <stdio.h>
void main() {
    float P, R, T, SI;
    printf("Enter Principal, Rate, Time: ");
    scanf("%f %f %f",&P,&R,&T);
    SI = (P*R*T)/100;
    printf("Simple Interest: %.2f", SI);
}
